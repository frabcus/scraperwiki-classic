<?php
    $sourcescraper = '{{sourcescraper}}';
    $keys = scraperwiki::getKeys($sourcescraper);
    $keyindex = getenv("URLQUERY");
    if ($keyindex == False)
        $keyindex = 4;
    $key = $keys[$keyindex];
    $counts = array();
    $s = scraperwiki::getData($sourcescraper, $limit=250);
    foreach ($s as $c => $row)
    {
        foreach ($row as $k => $value)
        {
            if ($k == $key)
            {
                if (!array_key_exists($value, $counts))
                    $counts[$value] =  0;
                $counts[$value] += 1;
            }
        }
    }
?>
